<!DOCTYPE html>

<html lang="zh-TW">

<head>

    <meta charset="UTF-8">

    <title>專案導覽</title>

</head>

<body>

    <h1>專案目錄</h1>

    <ul id="project-list">專案載入中...</ul>


    <script>

        // 請把這兩個變數換成你的 GitHub 帳號和儲存庫名稱

        const USERNAME = '你的_GITHUB_帳號';

        const REPO = '你的_REPOSITORY_名稱';

        

        async function fetchProjects() {

            const listElement = document.getElementById('project-list');

            try {

                // 向 GitHub API 請求根目錄的檔案清單

                const response = await fetch(`https://api.github.com/repos/${USERNAME}/${REPO}/contents/`);

                const data = await response.json();

                

                listElement.innerHTML = ''; // 清除載入中文字


                // 篩選出所有「資料夾」，並排除以點 (.) 開頭的隱藏資料夾

                const folders = data.filter(item => item.type === 'dir' && !item.name.startsWith('.'));


                if (folders.length === 0) {

                    listElement.innerHTML = '<li>目前沒有任何專案。</li>';

                    return;

                }


                // 自動生成 <li> 連結並塞入網頁

                folders.forEach(folder => {

                    const li = document.createElement('li');

                    li.innerHTML = `<a href="./${folder.name}/index.html">📂 ${folder.name}</a>`;

                    listElement.appendChild(li);

                });

            } catch (error) {

                console.error('抓取資料失敗:', error);

                listElement.innerHTML = '<li>無法載入專案列表。</li>';

            }

        }


        fetchProjects();

    </script>

</body>

</html>
